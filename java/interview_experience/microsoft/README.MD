public int evaluate(String exp){
	if(exp.isEmpty()) return 0;

	Stack<Character> st = new Stack<>();

	for(int i=0;i<exp.length();i++){
		char c = exp.charAt(i);
		if(c == '(' || Character.isDigit(c) || c == '-') || c == '+'){
			st.push(c);
		} else (c == ')'){
			int ans = 0;
			while(st.peek()!='('){
				char c = st.pop();
				if(Character.isDigit(c)){
					ans+= c;
				} else if(c=='-'){
					ans = -1 * ans;
				}
			}
			st.pop();
			st.push(Character.valueOf(ans));
		}
	}
	return st.pop();
}
